---

## **Usage**

```tsx
import React from "react";
import { Button, Card, Input, Badge } from "youruikit";
import "youruikit/dist/index.css";

function App() {
  return (
    <div className="p-4">
      <Card>
        <h1>Hello from YourUIKit!</h1>
        <Input placeholder="Type here..." />
        <Button>Click Me</Button>
        <Badge>New</Badge>
      </Card>
    </div>
  );
}

export default App;
```
